44 bool CIniFile::ReadFile()
45 {
46 
47  //fstream f;
48  ifstream f;
49  string line;
50  string keyname, valuename, value;
51  string::size_type pLeft, pRight;
52 
53 
54 
55  f.open(path.c_str(), ios::in);
56  if (f.fail())
57  return false;
58 
59  while (getline(f, line))
60  {
61  if (line.length())
62  {
63 
64  if (!isprint(line[0]))
65  {
66  printf("Failing on char %d\n", line[0]);
67  f.close();
68  return false;
69  }
70  if ((pLeft = line.find_first_of(";#[=")) != string::npos)
71  {
72  switch (line[pLeft])
73  {
74  case '[':
75  if ((pRight = line.find_last_of("]")) != string::npos &&
76  pRight > pLeft)
77  {
78  keyname = line.substr(pLeft + 1, pRight - pLeft - 1);
79  AddKeyName(keyname);
80  }
81  break;
82 
83  case '=':
84  valuename = line.substr(0, pLeft);
85  value = line.substr(pLeft + 1);
86 //#ifdef _DEBUG
87 // cout <<" VN: "<<valuename<<" V: "<<value<<endl;
88 //#endif
89  SetValue(keyname, valuename, value);
90  break;
91 
92  case ';':
93  case '#':
94  if (!names.size())
95  HeaderComment(line.substr(pLeft + 1));
96  else
97  KeyComment(keyname, line.substr(pLeft + 1));
98  break;
99  }
100  }
101  }
102  }
103 
104  f.close();
105  if (names.size())
106  return true;
107  return false;
108 }
