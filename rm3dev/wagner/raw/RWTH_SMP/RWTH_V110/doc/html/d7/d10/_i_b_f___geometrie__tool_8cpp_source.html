26  const kreis_t& k_In ,
27  const ellipse_t& e_In )
28 {
29  // Kreisgleichung k
30  double k_a00 = pow( k_In.r , -2 ) , // 1/r^2
31  k_a11 = pow( k_In.r , -2 ) , // 1/r^2
32  k_b0 = -2.0F * k_In.M.x * k_a00 , // -2xo/r^2
33  k_b1 = -2.0F * k_In.M.y * k_a11 , // -2yo/r^2
34  k_c = pow( k_In.M.x / k_In.r , 2 ) + pow( k_In.M.y / k_In.r , 2 ) - 1.0F ; // xo^2/r^2 + yo^2/r^2 - 1
35 
36  // Ellipsengleichung e
37  double e_a00 = pow( e_In.a , -2 ) , // 1/a^2
38  e_a11 = pow( e_In.b , -2 ) , // 1/b^2
39  e_b0 = -2.0F * e_In.M.x * e_a00 , // -2xo/a^2
40  e_b1 = -2.0F * e_In.M.y * e_a11 , // -2yo/b^2
41  e_c = pow( e_In.M.x / e_In.a , 2 ) + pow( e_In.M.y / e_In.b , 2 ) - 1.0F ; // xo^2/a^2 + yo^2/b^2 - 1
42 
43  double v1 = k_a00 * e_a11- e_a00 * k_a11,
44  v2 = k_a00 * e_b0 - e_a00 * k_b0 ,
45  v3 = k_a00 * e_b1 - e_a00 * k_b1 ,
46  v4 = k_a00 * e_c - e_a00 * k_c ,
47  v5 = k_a11 * e_b0 - e_a11 * k_b0 ,
48  v6 = k_b0 * e_b1 - e_b0 * k_b1 ,
49  v7 = k_b0 * e_c - e_b0 * k_c ;
50 
51  d5 u = { v2 * v7 - pow( v4 , 2 ) ,
52  v2 * v6 - 2.0F * v3 * v4 ,
53  -v5 * v2 - pow( v3 , 2 ) - 2.0F * v1 * v4 ,
54  -2.0F * v1 * v3 ,
55  -pow( v1 , 2 ) };
56 
57  d4 nst = {};
58  int znst = LF4( nst, u );
59 
60  if ( !znst ) return 0; // keine NSt (Schnittpunkte) gefunden
61 
62  d3 a = {},
63  b = {};
64  d2 nsta = {},
65  nstb = {};
66 
67  for ( int i=0; i < znst; i++ )
68  {
69  SPOut[i].y = (float)nst[i];
70 
71  a[0] = k_c + nst[i] * k_b1 + SPOut[i].y * nst[i] * k_a11;
72  a[1] = k_b0;
73  a[2] = k_a00;
74 
75  b[0] = e_c + nst[i] * e_b1 + SPOut[i].y * nst[i] * e_a11;
76  b[1] = e_b0;
77  b[2] = e_a00;
78 
79  LF2( nsta , a );
80  LF2( nstb , b );
81

83  else SPOut[i].x = (float)nsta[1];
84  }
85

90 // Berechnet die Schnittpunkte einer Gerade und einer Ellipse
91 int Schnittpunkt( P2& SPOut ,
92  const gerade2D_t& g_In ,
93  const ellipse_t& e_In )
94 {
95  // Geradengleichung mx + c = y
96  double g_b0 = ( g_In.Vr.dy ) / ( g_In.Vr.dx ) , // dy/dx = m
97  g_b1 = -1.0F , // -1y
98  g_c = g_In.Pu.y - ( g_b0 * g_In.Pu.x ); // yo - m*xo = c
99 
100  // Ellipsengleichung
101  double e_a00 = pow( e_In.a , -2 ) , // 1/a^2
102  e_a11 = pow( e_In.b , -2 ) , // 1/b^2
103  e_b0 = -2.0F * e_In.M.x * e_a00 , // -2xo/a^2
104  e_b1 = -2.0F * e_In.M.y * e_a11 , // -2yo/b^2
105  e_c = pow( e_In.M.x / e_In.a , 2 ) + pow( e_In.M.y / e_In.b , 2) - 1.0F ; // xo^2/a^2 + yo^2/b^2 - 1
106 
107  double v0 = -e_a00 * g_b0 ,
108  v1 = e_a00 ,
109  v2 = -e_a00 * g_c ,
110  v3 = e_a11 * g_b0 ,
111  v4 = g_b0 * e_b1 - e_b0 * g_b1 ,
112  v5 = g_b0 * e_c - e_b0 * g_c ;
113 
114  // Polynomkoeffizienten
115  d3 u = { v0 * v5 - pow( v2 , 2 ) ,
116  v0 * v4 - 2.0F * v1 * v2 ,
117  v0 * v3 - pow( v1 , 2 ) };
118 
119  // Nullstellen
120  d2 nst = {};
121 
122  int znst = LF2( nst , u );
123 
124  if ( znst > 0 ) // Schnittpunkte gefunden
125  {
126  SPOut[0].y = (float)nst[0];
127  SPOut[1].y = (float)nst[1];
128  SPOut[0].x = (float)( nst[0] - g_c ) / (float)g_b0 ;
129  SPOut[1].x = (float)( nst[1] - g_c ) / (float)g_b0 ;
130 
131  return znst;
132  };
133 
134  return 0; // keine Schnittpunkte
135 };

138 // Nullstellen eines Polynoms 4. Grades bestimmen (NUR reelle NSt)
139 // in Anlehnung an das Bairstow-Verfahren
140 int LF4(d4& nst, const d5& pn)
141 {
142  // Lokale Kopie des Eingabepolynoms, (wg. Schreibzugriff)
143  d5 poly = { pn[0] , pn[1] , pn[2] , pn[3] , pn[4] };
144 
145  // Maximale Anzahl an Iterationen
146  int maxiter = 20;
147 
148  // Anzahl der reellen Nullstellen (Returnvalue)
149  int znst = 0;
150 
151  int n = norm(poly), // Polynom normieren und Grad (n) bestimmen
152  j = n, // Schleifenzaehler
153  i = 0; // Schleifenzaehler
154 
155  if ( n > 2 ) // wenn poly eigentlich 2.Grades -> Abk¸rzung nehmen.
156  {
157  double b0 = poly[n], // Divisionsrest
158  b1 = 0,
159  b2;
160 
161  double q0 = 0, // Quotient
162  q1 = 0,
163  q2;
164 
165  double a0 = poly[n-2], // Startwert der Iteration
166  a1 = poly[n-1], // Startwert der Iteration
167  da0 = 0,
168  da1 = 0;
169 
170  double M = 0,
171  D = 0;
172 
173  do // Zerlegung von poly^n -> a^2 * b^(n-2)
174  {
175  b0 = poly[n];
176  b1 = 0;
177 
178  q0 = 0;
179  q1 = 0;
180 
181  for ( j = n - 1; j >= 0; j-- )
182  {
183  b2 = b1;
184  b1 = b0;
185  b0 = poly[j] - a1 * b1 - a0 * b2;
186  q2 = q1;
187  q1 = q0;
188  q0 = b2 - a1 * q1 - a0 * q2;
189  }
190 
191  M = -a0 * q1 - a1 * q0;
192  D = q0 * q0 - M * q1;
193 
194  da1 = (b0 * q1 - b1 * q0) / D;
195  da0 = (b1 * M - b0 * q0) / D;
196 
197  a1 = a1 - da1;
198  a0 = a0 - da0;
199  maxiter -=1 ;
200  }
201  while ( ( ( fabs(da0) > 0.0001F ) && ( fabs(da1) > 0.0001F ) ) || ( maxiter < 0 ) ); // Abbruchbedingung der Iterationsschleife
202 
203  d3 a = { a0, a1, 1 }; // P(2)
204  d3 b = {}; // P(n-2)
205 
206  // Eingabe Polynom reduzieren durch abdividieren der
207  // gefundenen Nullstellen ( poly(x)/a(x) = b(x) )
208  for( i = n-2 ; i >= 0 ; i-- )
209  {
210  b[i] = poly[i+2] / a[2];
211  for( j = 2 ; j >= 0 ; j-- )
212  poly[i+j] = poly[i + j] - a[j] * b[i];
213  }
214 
215  d2 n01 = {}; // x0,x1
216  d2 n23 = {}; // x2,x3
217 
218  // Nullstellen-Array initialisieren...
219  for ( i = 4 ; i > 0 ; nst[--i] = 0.0F );
220 
221  // Reelle Loesungen der quadratischen Faktoren bestimmen
222  znst = LF2( n01 , a );
223  if (znst)
224  {
225  nst[i++] = n01[0];
226  nst[i++] = n01[1];
227  }
228 
229  znst += LF2( n23 , b );
230  if (znst)
231  {
232  nst[i++] = n23[0];
233  nst[i ] = n23[1];
234  }
235  } // if (n>2)
236  else // Loesen alles <= 2.Grades
237  {
238  d3 a = { poly[0], poly[1], poly[2] };
239  d2 n01 = {};
240  znst = LF2( n01 , a );
241 
242  for ( i = 0 ; i < znst ; nst[i++] = n01[i] );
243  }
244 
245  return znst; // Rueckgabewert ist die Anzahl der Reellen Loesungen (0 bis 4)
246 };

249 // Bestimmung der REELLEN Loesungen einer quadratischen/linearen Gleichung
250 int LF2(d2& nst, const d3& pn)
251 {
252  double p,
253  q,
254  D;
255 
256  // Lokale Kopie des Eingabepolynoms, mit Schreibzugriff
257  d3 poly = { pn[0] , pn[1] , pn[2] };
258 
259  // Anzahl der reellen Nullstellen (Returnvalue)
260  int znst = 0;
261 
262  switch ( norm(poly) )
263  {
264  case 2: // Quadratische Gl. ->Loesen mit pq-Formel
265 
266  q = poly[0] / poly[2];
267  p = (poly[1] / poly[2]) / 2.0F; // genauer: p/2
268 
269  D = p*p - q;
270 
271  if (D > 0.0F) // zwei reelle Loesungen
272  {
273  nst[0] = -p+sqrt(D);
274  nst[1] = -p-sqrt(D);
275 
276  znst = 2;
277  }
278  else if ( D == 0.0F ) // doppelte Nullstelle
279  {
280  nst[0] = nst[1] = -p;
281 
282  znst = 2;
283  }
284  else // keine reelle Loesung
285  {
286  nst[0] = 0.0F;
287  nst[1] = 0.0F;
288 
289  znst = 0;
290  }
291  break;
292 
293  case 1: // Lineare Gl. -> Eine Loesung
294 
295  nst[0] = -poly[0] / poly[1];
296  nst[1] = 0.0F;
297 
298  znst = 1;
299  break;
300 
301  case 0: // Konstante -> keine Loesung oder 0
302 
303  nst[0] = 0.0F;
304  nst[1] = 0.0F;
305 
306  znst = ( poly[0] == 0.0F ) ? 1 : 0 ;
307  break;
308 
309  default: break;
310  };
311 
312  return znst; // Rueckgabewert ist die Anzahl der Reellen Loesungen (0 bis 2)
313 };
