22 Def_IBF_Ist_ZER ZER_trans_IST(const Def_Ist_Inter& Ist_Inter , const Def_IBF_Allgemein_IST& Ist_IBF)
23 {
24  Def_IBF_Ist_ZER ret = {};
25  const Def_Schuessel_Walzform& WForm = Ist_IBF.Schuessel.Walzform ;
26 
27  float phi_A,
28  phi_E;
29 
30  float x_a,y_a,
31  x_e,y_e;
32 
33  ret.ZRE.MTS = Ist_Inter.zal_pos, // MTSwerte Ein- und Auslaufseite
34  ret.ZRA.MTS = Ist_Inter.zar_pos;
35 
36  // Berechnung der Winkelpositionen der Zentrierarme
37  phi_E = phi_von_MTS( ret.ZRE.MTS );
38  phi_A = phi_von_MTS( ret.ZRA.MTS );
39 
40  x_a = -1074+1330*sin(phi_A);
41  y_a = -870 +1330*cos(phi_A);
42  x_e = -1074+1330*sin(phi_E);
43  y_e = 870 -1330*cos(phi_E);
44 
45  P4 SP_a = {},
46  SP_e = {};
47  kreis_t zre = { { x_e , y_e } , 135.0F },
48  zra = { { x_a , y_a } , 135.0F };
49 
50  ret.ZRA.Flag_Kontakt = Schnittpunkt(SP_a, zra, WForm.RingGeo);
51  ret.ZRE.Flag_Kontakt = Schnittpunkt(SP_e, zre, WForm.RingGeo);
52 
53  // Distanz[mm] der Zentrierollen zueinander
54  // dient hauptsaechlich der Kalibrierung
55  ret.delta_ZRA_ZRE = sqrt(pow((ret.ZRE.Pos_y = y_e)-(ret.ZRA.Pos_y = y_a),2)+pow(x_e-x_a,2)) - 270;
56 
57  // Tangentenwinkel im Kontaktpunkt Zentrierolle|Ring
58  // zur Berechnung der Kraftkomponenten auf den Ring
59  float phi_ri_ra_e = atan( ( ( WForm.RingGeo.b * x_e ) / WForm.RingGeo.a ) / ( ( -WForm.RingGeo.a * y_e ) / WForm.RingGeo.b)),
60  phi_ri_ra_a = atan( ( ( WForm.RingGeo.b * x_a ) / WForm.RingGeo.a ) / ( ( -WForm.RingGeo.a * y_a ) / WForm.RingGeo.b)),
61  phi_ra_ta_e = -phi_ri_ra_e - (PI/2) + phi_E,
62  phi_ra_ta_a = -phi_ri_ra_a - (PI/2) - phi_A;//+
63 
64  float Frad_e = -cos(phi_ra_ta_e) * Ist_Inter.zal_f * (560.0F/1330.0F),
65  Frad_a = cos(phi_ra_ta_a) * Ist_Inter.zar_f * (560.0F/1330.0F);
66 
67  // Kraftkomponenten in xy-Koordinaten
68  ret.ZRE.F_y = Frad_e * cos(phi_ri_ra_e);
69  ret.ZRE.F_x = Frad_e * sin(phi_ri_ra_e);
70  ret.ZRA.F_y = Frad_a * cos(phi_ra_ta_a);
71  ret.ZRA.F_x = Frad_a * -sin(phi_ra_ta_a);
72 
73  return ret;
74 };

77 void ZER_trans_Soll(struct Def_Soll_Inter& Soll_Inter, const Def_IBF_Allgemein_SOLL& Soll_IBF,const Def_IBF_Allgemein_IST& Ist_IBF)
78 {
79  const Def_Schuessel_Walzform& WForm = Ist_IBF.Schuessel.Walzform ;
80 
81  kreis_t ZArm_E = { {-1074, 870} , // Lagerkoordinaten Einlaufseite
82  1330 }; // Radius Lager_arm -> Lager_Rolle
83 
84  kreis_t ZArm_A = { {-1074, -870} , // Lagerkoordinaten Auslaufseite
85  1330 }; // Radius Lager_arm -> Lager_Rolle
86 
87  ellipse_t Ring_A = { { WForm.RingGeo.M.x,
88  WForm.RingGeo.M.y } ,
89  WForm.RingGeo.a + Soll_IBF.ZER_Param.ZRA.delta_Ring + 135,
90  WForm.RingGeo.b + Soll_IBF.ZER_Param.ZRA.delta_Ring + 135 };
91 
92  ellipse_t Ring_E = { { WForm.RingGeo.M.x,
93  WForm.RingGeo.M.y } ,
94  WForm.RingGeo.a + Soll_IBF.ZER_Param.ZRE.delta_Ring + 135,
95  WForm.RingGeo.b + Soll_IBF.ZER_Param.ZRE.delta_Ring + 135 };
96 
97  P4 SP_E = {},
98  SP_A = {};
99 
100  float phi_E = 0.0F,
101  phi_A = 0.0F;
102 
103  if (!Schnittpunkt(SP_E , ZArm_E, Ring_E))
104  {
105  // Kein Schnittpunkt! -> Position halten.
106  Soll_Inter.zal_pos = Ist_IBF.ZER_Param.ZRE.MTS;
107  }
108  else
109  {
110  phi_E = (SP_E[0].y>SP_E[1].y) ? acos( ( ZArm_E.M.y - SP_E[0].y ) / 1330 ) : acos( ( ZArm_E.M.y - SP_E[1].y ) / 1330 );
111  Soll_Inter.zal_pos = MTS_von_phi( phi_E );
112  };
113 
114  if (!Schnittpunkt(SP_A , ZArm_A, Ring_A))
115  {
116  // Kein Schnittpunkt! -> Position halten.
117  Soll_Inter.zar_pos = Ist_IBF.ZER_Param.ZRA.MTS;
118  }
119  else
120  {
121  phi_A = (SP_A[0].y<SP_A[1].y) ? acos( ( -ZArm_A.M.y + SP_A[0].y ) / 1330 ) : acos( ( -ZArm_A.M.y + SP_A[1].y ) / 1330 );
122  Soll_Inter.zar_pos = MTS_von_phi( phi_A );
123  };
124  Soll_Inter.zal_pos = Soll_Inter.zar_pos;
125  Soll_Inter.zar_v = Soll_Inter.zal_v = Soll_IBF.ZER_Param.ZRE.V_Spur * ( 560.0F / 1330.0F );
126 
127  // Modus und Steuerung (act 0=SMS/1=IBF)
128  Soll_Inter.zen_mode = Soll_IBF.ZER_Param.mode;
129  Soll_Inter.zen_act = Soll_IBF.ZER_Param.act;
130 
131  return;
132 };
