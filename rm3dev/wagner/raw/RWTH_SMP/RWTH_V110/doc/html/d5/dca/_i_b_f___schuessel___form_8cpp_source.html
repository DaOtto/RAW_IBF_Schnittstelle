20 int Schuessel_Form_IST( Def_IBF_Allgemein_IST& Ist_IBF,
21  const Def_Ist_Inter& Ist_Inter )
22 {
23  Def_Schuessel_Walzform& WF = Ist_IBF.Schuessel.Walzform ;
24 
25  WF.da = Ist_Inter.tr_pos+Ist_Inter.ag_pos_real-Ist_Inter.Werkzeug.l_fehlende_spitze;
26 
27  // Bei Testfahrt ohne Ring, Tastrollenvorgabe aus IBF_Data.ini verwenden
28  if (Ist_IBF.Schuessel.Test.Flag_oRing) WF.da = Ist_IBF.Schuessel.Test.Tastrolle_da;
29 
30  // Simuliertes Ringwachstum im Testmodus
31  if (Ist_IBF.Schuessel.Test.Flag_oRing) WF.da = WF.da * (1 + Ist_IBF.K_Iter*0.0003F);
32 
33 
35  // Puffern und filtern der Sensorwerte durch Mittelwertbildung
37 
38  WF.cursor = Ist_IBF.Iter % PUFF_LEN;
39  WF.TR_puffer[WF.cursor] = WF.da;
40  WF.DRx_puffer[WF.cursor] = Ist_IBF.DRR_Param.DrR_E.PosXY.x;
41  WF.DRy_puffer[WF.cursor] = Ist_IBF.DRR_Param.DrR_E.PosXY.y;
42 
43  WF.da_mf = 0.0F;
44  WF.DRx_mf = 0.0F;
45  WF.DRy_mf = 0.0F;
46 
47  for ( int i = PUFF_LEN ; --i >= 0 ; )
48  {
49  WF.da_mf += WF.TR_puffer[i];
50  WF.DRx_mf += WF.DRx_puffer[i];
51  WF.DRy_mf += WF.DRy_puffer[i];
52  }
53 
54  WF.da_mf = WF.da_mf / PUFF_LEN;
55  WF.DRx_mf = WF.DRx_mf / PUFF_LEN;
56  WF.DRy_mf = WF.DRy_mf / PUFF_LEN;
57 
59  // Berechnung der Radien der Projektion des Ringes (Schuessel) in der Walztischebene
61 
62  WF.RingGeo.a = WF.da_mf / 2.0F; // Halbachse 'a' der elliptischen Ringprojektion (Laengs)
63 
64  // Berechnung der langen Halbachse b, wenn Druckrolle Kontakt zum Ring hat.
65  if(Ist_IBF.Flag.DRE_kraft_anfahren_erreicht)
66  {
67  float al = asin( ( WF.DRx_mf / WF.RingGeo.a ) - 1.0F );
68  WF.RingGeo.b = fabs( WF.DRy_mf / cos( al ) ); // Halbachse 'b' der elliptischen Ringprojektion (Quer)
69  }
70  else
71  {
72  WF.RingGeo.b = WF.RingGeo.a;// * 1.065F; // 20 Kletterwinkel
73 
74  // Simuliertes Ringklettern im Testmodus
75  if (Ist_IBF.Schuessel.Test.Flag_oRing) WF.RingGeo.b = WF.RingGeo.a * (1 + min(0.065F,Ist_IBF.K_Iter*0.0001F));
76  }
77 
78 
80  // Festlegen der Lage der Ringprojektion auf die Walztischebene
82 
83  WF.RingGeo.M.x = WF.da_mf/2.0F; // Mittelpunkt X
84  WF.RingGeo.M.y = 0.0; // Mittelpunkt Y
85 
87  // Berechnung des aktuellen Kletterwinkels
88  // Verhaeltnis der Radien a/b. Wird fuer einige berechnungen benoetigt, und hier vorberechnet
90 
91  float a_zu_b = WF.RingGeo.a / WF.RingGeo.b;
92  WF.Kletterwinkel = acos(min(1.0F,a_zu_b));
93 
95  //Berechnung der momentanen Wandbreite sm des Ringprofils
97 
98  // Wanddicke sm des Ringprofils aus Axialwalzspalthoehe, Volumenkonstanz und Kletterwinkel berechnen
99 
100  WF.hm = min(Ist_IBF.ASO_Pos , Ist_IBF.Schuessel.Vorform.hm_mm); // Aktuelle Hoehe des Ringprofils
101  d3 a = {}; // double[3] , Koeffizientenvektor
102  d2 l = {}; // double[2] , Loesungsvektor
103 
104  //a[2] = -PI*WF.hm*pow(cos(WF.Kletterwinkel),2);
105  //a[1] = 2.0F*PI*WF.hm*WF.RingGeo.b*cos(WF.Kletterwinkel) - PI*pow(WF.hm,2)*sin(WF.Kletterwinkel)*cos(WF.Kletterwinkel);
106  //a[0] = -Ist_IBF.Schuessel.Vorform.Volumen;
107 
108  a[2] = a_zu_b*a_zu_b;
109  a[1] = WF.hm*sin(WF.Kletterwinkel)*a_zu_b - WF.da_mf;
110  a[0] = Ist_IBF.Schuessel.Vorform.Volumen / (PI*WF.hm);
111 
112  LF2( l , a );
113 
114  // die relevante Loesung ist die betragsmaessig kleinere ->
115  WF.sm = (float)min( l[0] , l[1] );
116 
118  // Berechnung der aktuellen Kletterhoehe
120 
121  WF.Kletterhoehe = 2.0F * WF.RingGeo.b * sin( WF.Kletterwinkel ) + WF.hm;
122 
124  // Positionsbereich Dornwalze
126 
127  WF.dw_min_Pos = WF.sm * cos( 2.0F * WF.Kletterwinkel ) + WF.hm * sin( 2.0F * WF.Kletterwinkel );
128  WF.dw_max_Pos = WF.da_mf - WF.sm - Ist_Inter.Werkzeug.Durchmesser_DW;
129 
131  // Effektive Radien zur Berechnung der Axialwalzendrehzahlen
133 
134  WF.r_eff_u = WF.RingGeo.b;
135  WF.r_eff_o = WF.r_eff_u - WF.hm * sin( WF.Kletterwinkel );
136 
137  return 0;
138 };

140 int Vorform_geo_berechnen( Def_IBF_Allgemein_IST& Ist_IBF )
141 {
142  // Vorformvolumen berechnen in [mm^3]. Dichte und Masse sind in IBF_Data.ini definiert.
143  Ist_IBF.Schuessel.Vorform.Volumen = Ist_IBF.Schuessel.Vorform.Masse_kg / Ist_IBF.Schuessel.Vorform.Dichte_warm * 1000000.0F;
144 
145  // Mittlere Profilbreite [mm] berechnen.
146  float di = sqrt(pow(Ist_IBF.Schuessel.Vorform.da_mm,2)-Ist_IBF.Schuessel.Vorform.Volumen/(PI*Ist_IBF.Schuessel.Vorform.hm_mm/4.0F));
147 
148  Ist_IBF.Schuessel.Vorform.sm_mm = (Ist_IBF.Schuessel.Vorform.da_mm - di) / 2.0F;
149 
150  return 0;
151 };
